<span class="vocabulary">**coffee**</span> ['kɒfɪ]

<span class="vocabulary">**chocolate**</span> ['tʃɒkəlɪt]

<span class="vocabulary">**tea**</span> [ti:]

<span class="vocabulary">**orange**</span> ['ɒrɪndӡ]

<span class="vocabulary">**lemon**</span> ['lemən]
